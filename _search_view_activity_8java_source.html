304 private double inRange(double lat, double lon, double lat2, double lon2) {
305 double earthRadius = 6371;
306 double dLat = (lat - lat2) * Math.PI / 180.0;
307 double dLon = (lon - lon2) * Math.PI / 180.0;
308 double a = Math.sin(dLat / 2.) * Math.sin(dLat / 2.)
309 +Math.cos(lat * Math.PI / 180.0)
310 * Math.cos(lat2 * Math.PI / 180.0)
311 * Math.sin(dLon / 2.) * Math.sin(dLon / 2.);
312 double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1. -a));
313 double d = earthRadius * c;
314 //Log.i("distance " + d, " Hellbach");
315 //Log.i("sliderdistance " + sliderDistance, " Hellbach");
316 return d;
317 }
