246 private double inRange(double lat, double lon) {
247 double earthRadius = 6371;
248 double dLat = (lat - userLat) * Math.PI / 180.0;
249 double dLon = (lon - userLon) * Math.PI / 180.0;
250 double a = Math.sin(dLat / 2.) * Math.sin(dLat / 2.)
251 +Math.cos(lat * Math.PI / 180.0)
252 * Math.cos(userLat * Math.PI / 180.0)
253 * Math.sin(dLon / 2.) * Math.sin(dLon / 2.);
254 double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1. -a));
255 double d = earthRadius * c;
256 //Log.i("distance " + d, " Hellbach");
257 //Log.i("sliderdistance " + sliderDistance, " Hellbach");
258 return d;
259 }
