42 if (ActivityCompat.shouldShowRequestPermissionRationale(activity, permission)) {
43 // Display a dialog with rationale.

64 public static boolean isPermissionGranted(String[] grantPermissions, int[] grantResults,
65 String permission) {
66 for (int i = 0; i < grantPermissions.length; i++) {
67 if (permission.equals(grantPermissions[i])) {
68 return grantResults[i] == PackageManager.PERMISSION_GRANTED;
69 }
70 }
71 return false;

155 public static RationaleDialog newInstance(int requestCode, boolean finishActivity) {

169 @Override
170 public Dialog onCreateDialog(Bundle savedInstanceState) {
171 Bundle arguments = getArguments();
172 final int requestCode = arguments.getInt(ARGUMENT_PERMISSION_REQUEST_CODE);
173 finishActivity = arguments.getBoolean(ARGUMENT_FINISH_ACTIVITY);
174
175 return new AlertDialog.Builder(getActivity())
176 .setMessage("Permission Location")
177 .setPositiveButton(android.R.string.ok, new DialogInterface.OnClickListener() {
178 @Override
179 public void onClick(DialogInterface dialog, int which) {
180 // After click on Ok, request the permission.
181 ActivityCompat.requestPermissions(getActivity(),
182 new String[]{Manifest.permission.ACCESS_FINE_LOCATION},
183 requestCode);
184 // Do not finish the Activity while requesting permission.
185 finishActivity = false;
186 }
187 })
188 .setNegativeButton(android.R.string.cancel, null)
189 .create();
190 }
191

197 public void onDismiss(DialogInterface dialog) {
198 super.onDismiss(dialog);
199 if (finishActivity) {
200 Toast.makeText(getActivity(),
